    // 三路快排
    void q3way(int[] arr, int lo, int hi) {
        if (hi <= lo) {
            return;
        }
        int lt = lo;
        int i = lt + 1;
        int gt = hi;
        int pivotal = arr[lo];
        while (i <= gt) {
            if (arr[i] < pivotal) {
                 //目标元素小于中轴元素，与低位交换后，元素都已检查，故各自增1
                exch(arr, lt++, i++);
            } else if (arr[i] > pivotal) {
                //目标元素大于中轴元素，与高位位交换后，高位减1。因为换过来的元素还未检查，不能自增
                exch(arr, i, gt--);
            } else {
                //与中轴元素相等，不用交换，直接下一个
                i++;
            }
        }
        q3way(arr, lo, lt - 1);
        q3way(arr, gt + 1, hi);
    }
    //快排
    void qsort(int[] arr, int lo, int hi) {
        if (lo >= hi) {
            return;
        }
        int j = PARTITION(arr, lo, hi);
        qsort(arr, lo, j - 1);
        qsort(arr, j + 1, hi);
    }
    //洗牌算法，避免快排的最坏情况
    void shuffle(int[] arr) {
        Random r = new Random();
        for (int i = arr.length; i > 1; i--) {
            int j = r.nextInt(i);
            exch(arr, i - 1, j);
        }
    }
    // 找到第K大的数
    int FIND(int[] a, int k) {
        int lo = 0, hi = a.length - 1;
        while (lo < hi) {
            int j = PARTITION(a, lo, hi);
            if (j < k) {
                lo = j + 1;
            } else if (j > k) {
                hi = j - 1;
            } else {
                return a[k];
            }
        }
        return a[k];
    }
    // 数组元素在[lo,hi]之间，既包括左边界，又包括右边界
    int PARTITION(int[] a, int lo, int hi) {
        int i = lo, j = hi + 1;
        int pivatol = a[lo];
        while (true) {
            // 在不越界的情况下，找到一个元素，比 a[lo] 大，其数组下标为 i
            while (a[++i] < pivatol) {
                if (i == hi) {
                    break;
                }
            }
            // 在不越界的情况下，找到一个元素，比 a[lo] 小，其数组下标为 j
            while (pivatol < a[--j]) {
                if (j == lo) {
                    break;
                }
            }
            // 必须 i<j
            if (i >= j) {
                break;
            }
            // 将两个“逆序”的元素交换，使得小于a[lo]的元素都在左边，大于a[lo]的元素都在右边
            exch(a, i, j);
        }
        // 将 a[lo]元素置于数组中正确的位置
        exch(a, lo, j);
        // 返回 a[lo] 元素在数组中的位置
        return j;
    }
     // 交换
    void exch(int[] arr, int lo, int hi) {

        int tmp = arr[lo];
        arr[lo] = arr[hi];
        arr[hi] = tmp;
    }
