public class StringSort {
    // 低位优先的字符串排序
    // 通过前W个字符将a[] 排序
    public static void LSD(String[] a, final int W, final int R) {

        final int N = a.length;
        String[] aux = new String[N];
        // 根据第d个字符用键索引计数法排序
        for (int d = W - 1; d >= 0; d--) {
            // 需要计算累积和，长度+1，又某些短字符串没有字符，返回-1，为了不越界，再+1
            int[] count = new int[R + 2];
            // 计算频率（count 索引为 0 的值总是0，起始计数从下标1开始，即+1；又考虑索引为-1的情况，再+1）
            for (int i = 0; i <= N - 1; i++) {
                count[charAt(a[i], d) + 2]++;
            }
            // 将频率转化为索引
            for (int r = 0; r <= R; r++) {
                count[r + 1] += count[r];
            }
            // 数据分类
            for (int i = 0; i <= N - 1; i++) {
                aux[count[charAt(a[i], d) + 1]++] = a[i];
            }
            // 写回
            for (int i = 0; i <= N - 1; i++) {
                a[i] = aux[i];
            }
        }
    }
    // 高位优先的字符串排序
    public static void MSD(String[] a, final int R, final int M) {
        final int N = a.length;
        String[] aux = new String[N];
        MSDSort(a, aux, 0, N - 1, 0, R, M);
    }
    // 以第d个字符为键，将a[lo] 至 a[hi] 排序
    private static void MSDSort(String[] a, String[] aux, int lo, int hi, int d,
            final int R, final int M) {
        if (hi <= lo + M) {
            // 小数组优化，切换值插入排序
            return;
        }
        // 需要计算累积和，长度+1，又某些短字符串没有字符，返回-1，为了不越界，再+1
        int[] count = new int[R + 2];
        // 计算频率（count 索引为 0 的值总是0，起始计数从下标1开始，即+1；又考虑索引为-1的情况，再+1）
        for (int i = lo; i <= hi; i++) {
            count[charAt(a[i], d) + 2]++;
        }
        // 将频率转化为索引
        for (int r = 0; r <= R; r++) {
            count[r + 1] += count[r];
        }
        // 数据分类
        for (int i = lo; i <= hi; i++) {
            aux[count[charAt(a[i], d) + 1]++] = a[i];
        }
        // 写回
        for (int i = lo; i <= hi; i++) {
            a[i] = aux[i - lo];
        }
        // 递归的以每个字符作为键进行排序
        for (int r = 0; r < R; r++) {
            MSDSort(a, aux, lo + count[r], lo + count[r + 1] - 1, d + 1, R, M);
        }
    }

    // 三向字符串快速排序
    public static void Quick3string(String[] a) {
        q3sort(a, 0, a.length - 1, 0);
    }
    private static void q3sort(String[] a, int lo, int hi, int d) {
        if (hi <= lo) {
            return;
        }
        int lt = lo;
        int gt = hi;
        int v = charAt(a[lo], d);
        int i = lo + 1;
        while (i <= gt) {
            int t = charAt(a[i], d);
            if (t < v) {
                exch(a, lt++, i++);
            } else if (t > v) {
                exch(a, i, gt--);
            } else {
                i++;
            }
        }
        // a[lo...lt-1]< v= a[lt...gt] < a[gt+1,hi]
        q3sort(a, lo, lt - 1, d);
        if (v >= 0) {
            q3sort(a, gt + 1, hi, d);
        }
        q3sort(a, lt, gt, d + 1);
    }
    // 交换
    private static void exch(String[] arr, int lo, int hi) {

        String tmp = arr[lo];
        arr[lo] = arr[hi];
        arr[hi] = tmp;
    }

    private static int charAt(String s, int d) {
        if (d < s.length()) {
            return s.charAt(d);
        }
        return -1;
    }
}
